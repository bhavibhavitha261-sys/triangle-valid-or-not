#include <stdio.h>

int main() {
    int a, b, c;

    printf("Enter three sides of triangle: ");
    scanf("%d %d %d", &a, &b, &c);

    if (a + b > c && a + c > b && b + c > a) {
        printf("Triangle is VALID\n");
    } else {
        printf("Triangle is NOT VALID\n");
    }

    return 0;
}
